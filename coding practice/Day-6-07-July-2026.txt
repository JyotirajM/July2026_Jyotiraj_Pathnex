## Ansible - Install Docker

```yaml
- name: Install Docker
  hosts: all
  become: yes
  tasks:
    - name: Install Docker
      yum:
        name: docker
        state: present
    - name: Start Docker
      service: 
        name: docker
        state: started
        enabled: yes

# Kubernetes - pod with volume mount
apiVersion: v1
kind: Pod
metadata:
  name: pathnex-pod
spec:
  containers:
    - name: web
      image: nginx
      volumeMounts:
        - mountPath: /usr/share/nginx/html
          name: html-volume
   volumes:
     - name: html-volume
       hostpath:
         path: /data
         type: Directory

# Shell Script - Check for running services
#!/bin/bash
services=("nginx""docker""httpd")
for services in "${services[@]}"
do
  if  systemctl is -active --quiet $service; then
    echo "$service is running"
  else
    echo "$service is not running"
  fi
done
